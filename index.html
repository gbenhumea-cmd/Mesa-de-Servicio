// ══════════════════════════════════════════════════════ // CSCO — Mesa de Servicio — Apps Script Backend v3 // Google Sheet ID: 1PGtO2n13ZY2tO4ivaE7OqXhvQ6NiEAucUfqhl2jE0jg // ══════════════════════════════════════════════════════ const SHEET_ID = '1PGtO2n13ZY2tO4ivaE7OqXhvQ6NiEAucUfqhl2jE0jg'; const SHEET_TICKETS = 'Tickets'; const SHEET_NOTAS = 'Notas'; const SHEET_CONFIG = 'Config'; const COL_TKT = ['id','empresa','contacto','tel','email','canal','prioridad', 'dir','indic','supInstruc','desc','estatus','tecnico', 'tecnicoName','otFolio','otFecha','creadoPor','fecha']; const COL_NOTA = ['ticketId','tipo','texto','autor','fecha']; // ════════════════════════════════════════════════════ // INICIALIZAR — ejecutar una vez manualmente // ════════════════════════════════════════════════════ function inicializarHojas() { const ss = SpreadsheetApp.openById(SHEET_ID); let ht = ss.getSheetByName(SHEET_TICKETS); if (!ht) { ht = ss.insertSheet(SHEET_TICKETS); ht.appendRow(COL_TKT); ht.getRange(1,1,1,COL_TKT.length) .setFontWeight('bold').setBackground('#1a5276').setFontColor('#ffffff'); ht.setFrozenRows(1); } let hn = ss.getSheetByName(SHEET_NOTAS); if (!hn) { hn = ss.insertSheet(SHEET_NOTAS); hn.appendRow(COL_NOTA); hn.getRange(1,1,1,COL_NOTA.length) .setFontWeight('bold').setBackground('#1a5276').setFontColor('#ffffff'); hn.setFrozenRows(1); } let hc = ss.getSheetByName(SHEET_CONFIG); if (!hc) { hc = ss.insertSheet(SHEET_CONFIG); hc.appendRow(['clave','valor']); hc.appendRow(['folioNum','1']); } return 'OK'; } // ════════════════════════════════════════════════════ // UTILIDADES // ════════════════════════════════════════════════════ function getSheet(n) { return SpreadsheetApp.openById(SHEET_ID).getSheetByName(n); } function getConfig(clave) { const data = getSheet(SHEET_CONFIG).getDataRange().getValues(); for (let i = 1; i < data.length; i++) if (data[i][0] === clave) return data[i][1]; return 1; } function setConfig(clave, valor) { const hc = getSheet(SHEET_CONFIG); const data = hc.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === clave) { hc.getRange(i+1,2).setValue(valor); return; } } hc.appendRow([clave, valor]); } function nuevoFolio() { const n = parseInt(getConfig('folioNum') || 1); setConfig('folioNum', n + 1); return 'TKT-' + new Date().getFullYear() + '-' + String(n).padStart(4,'0'); } function ok(data) { return ContentService .createTextOutput(JSON.stringify({ok:true, data})) .setMimeType(ContentService.MimeType.JSON); } function err(msg) { return ContentService .createTextOutput(JSON.stringify({ok:false, error:msg})) .setMimeType(ContentService.MimeType.JSON); } // ════════════════════════════════════════════════════ // ROUTER — cada acción viaja como parámetro GET // ════════════════════════════════════════════════════ function doGet(e) { try { const action = e.parameter.action || ''; if (action === 'ping') return ok('pong'); if (action === 'getTickets') return getTickets(); // Escritura: los datos viajan en el parámetro 'd' como JSON codificado en URL if (action === 'crearTicket') { const ticket = JSON.parse(decodeURIComponent(e.parameter.d || '{}')); return crearTicket(ticket); } if (action === 'actualizarTicket') { const ticket = JSON.parse(decodeURIComponent(e.parameter.d || '{}')); return actualizarTicket(ticket); } if (action === 'agregarNota') { const p = JSON.parse(decodeURIComponent(e.parameter.d || '{}')); return agregarNota(p.ticketId, p.nota); } return err('Accion no reconocida: ' + action); } catch(ex) { return err(ex.message + ' | ' + ex.stack); } } // ════════════════════════════════════════════════════ // OPERACIONES // ════════════════════════════════════════════════════ function getTickets() { const ht = getSheet(SHEET_TICKETS); const hn = getSheet(SHEET_NOTAS); if (!ht || !hn) return err('Ejecuta inicializarHojas() primero'); const tRows = ht.getDataRange().getValues(); const nRows = hn.getDataRange().getValues(); const notasMap = {}; for (let i = 1; i < nRows.length; i++) { const [ticketId, tipo, texto, autor, fecha] = nRows[i]; if (!notasMap[ticketId]) notasMap[ticketId] = []; notasMap[ticketId].push({tipo, texto, autor, fecha}); } const tickets = []; for (let i = 1; i < tRows.length; i++) { const row = tRows[i]; if (!row[0]) continue; const t = {}; COL_TKT.forEach((col, idx) => t[col] = row[idx] || ''); t.notas = notasMap[t.id] || []; tickets.push(t); } return ok(tickets); } function crearTicket(ticket) { const ht = getSheet(SHEET_TICKETS); const hn = getSheet(SHEET_NOTAS); if (!ht || !hn) return err('Hojas no inicializadas'); ticket.id = nuevoFolio(); ht.appendRow(COL_TKT.map(col => ticket[col] || '')); if (ticket.notas && ticket.notas[0]) { const n = ticket.notas[0]; hn.appendRow([ticket.id, n.tipo, n.texto, n.autor, n.fecha]); } return ok({id: ticket.id}); } function actualizarTicket(ticket) { const ht = getSheet(SHEET_TICKETS); const data = ht.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket.id) { const row = COL_TKT.map((col, idx) => ticket[col] !== undefined ? ticket[col] : data[i][idx] ); ht.getRange(i+1, 1, 1, COL_TKT.length).setValues([row]); return ok({updated: ticket.id}); } } return err('Ticket no encontrado: ' + ticket.id); } function agregarNota(ticketId, nota) { getSheet(SHEET_NOTAS).appendRow( [ticketId, nota.tipo, nota.texto, nota.autor, nota.fecha] ); return ok({added: true}); }
